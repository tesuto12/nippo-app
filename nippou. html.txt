<script>
var proc='',numStr='',adjMs=0,records=[];
try{records=JSON.parse(localStorage.getItem('nippo_v71')||'[]');}catch(e){}

function todayStr(){var n=new Date();return n.getFullYear()+'/'+(n.getMonth()+1<10?'0':'')+(n.getMonth()+1)+'/'+(n.getDate()<10?'0':'')+n.getDate();}
function getTs(){return new Date(Date.now()+adjMs);}
function fmtDt(d){var dt=d instanceof Date?d:new Date(d);return dt.getFullYear()+'/'+(dt.getMonth()+1<10?'0':'')+(dt.getMonth()+1)+'/'+(dt.getDate()<10?'0':'')+dt.getDate()+' '+(dt.getHours()<10?'0':'')+dt.getHours()+':'+(dt.getMinutes()<10?'0':'')+dt.getMinutes();}
function fmtTime(d){var dt=d instanceof Date?d:new Date(d);return(dt.getHours()<10?'0':'')+dt.getHours()+':'+(dt.getMinutes()<10?'0':'')+dt.getMinutes();}
function updDt(){document.getElementById('dtVal').textContent='設定時刻: '+fmtTime(getTs())+'  (タップでリセット)';}
function adj(min){adjMs+=min*60000;updDt();showToast(min>0?'+'+min+'分':min+'分');}
function resetDt(){adjMs=0;updDt();showToast('現在時刻にリセット');}

function checkAutoSet(){
  var n=new Date();var h=n.getHours();var m=n.getMinutes();
  var today=todayStr();var key='autoset_'+today;
  var done=localStorage.getItem(key)||'';
  if(h===8&&m>=30&&m<=40&&done.indexOf('朝礼')<0){
    document.getElementById('autoBanner').textContent='朝礼を自動セットしました';
    document.getElementById('autoBanner').classList.add('show');
    autoSend('朝礼');localStorage.setItem(key,done+'朝礼');
  } else if(h===12&&m>=0&&m<=10&&done.indexOf('休憩')<0){
    document.getElementById('autoBanner').textContent='12:00 休憩を自動セットしました';
    document.getElementById('autoBanner').classList.add('show');
    autoSend('休憩');localStorage.setItem(key,done+'休憩');
  }
}

function sw(name,btn){
  document.querySelectorAll('.tab').forEach(function(t){t.classList.remove('active');});
  document.querySelectorAll('.pane').forEach(function(p){p.classList.remove('active');});
  btn.classList.add('active');document.getElementById('pane-'+name).classList.add('active');
  if(name==='history')renderTimeline();
}
function selProc(p,btn){
  proc=p;document.querySelectorAll('.pbtn').forEach(function(b){b.classList.remove('sel','oth');});
  btn.classList.add('sel');if(p==='その他')btn.classList.add('oth');
}
function kp(k){
  if(k==='C')numStr='';
  else if(k==='BS')numStr=numStr.slice(0,-1);
  else{if(numStr==='0')numStr='';numStr+=k;}
  var el=document.getElementById('numDisp');
  el.textContent=numStr||' ';el.classList.toggle('has-val',numStr.length>0);
}
function mkRec(p,qty,memo){var ts=getTs();return{id:Date.now()+Math.random(),ts:ts.getTime(),datetime:fmtDt(ts),process:p,qty:qty,memo:memo||''};}
function addRec(r){records.unshift(r);try{localStorage.setItem('nippo_v71',JSON.stringify(records.slice(-200)));}catch(e){}flash();}
function doSend(){if(!proc){showToast('工程を選んでください');return;}var qty=parseInt(numStr)||0;addRec(mkRec(proc,qty,''));showToast(proc+' / '+qty+'個');resetForm();}
function quickSend(){addRec(mkRec('その他',0,'クイック完了'));showToast('クイック完了');}
function autoSend(t){addRec(mkRec('その他',0,t));showToast(t+'を記録しました');}
function resetForm(){proc='';numStr='';adjMs=0;document.querySelectorAll('.pbtn').forEach(function(b){b.classList.remove('sel','oth');});document.getElementById('numDisp').textContent=' ';document.getElementById('numDisp').classList.remove('has-val');updDt();}

function renderTimeline(){
  var today=todayStr();
  var recs=records.filter(function(r){return r.datetime.indexOf(today)===0;}).slice().reverse();
  var total=0;recs.forEach(function(r){if(['切断','手加工','NC加工','組立'].indexOf(r.process)>=0)total+=r.qty;});
  document.getElementById('totalQty').textContent=total+' 個';
  var el=document.getElementById('timeline');
  if(!recs.length){el.innerHTML='<div class="empty">今日の記録はまだありません</div>';return;}
  el.innerHTML=recs.map(function(r){
    return '<div class="tl-item"><div style="flex:1;min-width:0"><div class="tl-time">'+r.datetime.split(' ')[1]+'</div><div class="tl-proc">'+r.process+'</div>'+(r.memo?'<div class="tl-memo">'+r.memo+'</div>':'')+'</div><div style="display:flex;align-items:center"><span class="tl-qty">'+(r.qty>0?r.qty+'個':'-')+'</span><button class="tl-del" onclick="delRec('+r.id+')">削除</button></div></div>';
  }).join('');
}
function delRec(id){records=records.filter(function(r){return r.id!==id;});try{localStorage.setItem('nippo_v71',JSON.stringify(records));}catch(e){}renderTimeline();showToast('削除しました');}

function copyExcel(){
  var today=todayStr();
  var recs=records.filter(function(r){return r.datetime.indexOf(today)===0;}).slice().reverse();
  if(!recs.length){showToast('今日の記録がありません');return;}
  var text='日付/時刻\t作業者\t工程\t数量\t備考\n'+recs.map(function(r){return r.datetime+'\t自分\t'+r.process+'\t'+r.qty+'\t'+(r.memo||'');}).join('\n');
  navigator.clipboard.writeText(text).then(function(){
    var btn=document.getElementById('copyBtn');btn.classList.add('done');btn.textContent='コピー完了！ パソコンでCtrl+V';
    showToast(recs.length+'件をコピーしました');
    setTimeout(function(){btn.classList.remove('done');btn.textContent='Excel用にコピー (パソコンでCtrl+V)';},3000);
  }).catch(function(){showToast('コピー失敗。ブラウザの許可を確認');});
}

function flash(){var e=document.getElementById('fl');e.classList.add('show');setTimeout(function(){e.classList.remove('show');},300);}
function showToast(msg){var e=document.getElementById('toast');e.textContent=msg;e.classList.add('show');setTimeout(function(){e.classList.remove('show');},2200);}
setInterval(function(){
  var n=new Date();
  document.getElementById('hTime').textContent=(n.getHours()<10?'0':'')+n.getHours()+':'+(n.getMinutes()<10?'0':'')+n.getMinutes()+':'+(n.getSeconds()<10?'0':'')+n.getSeconds();
  updDt();
},1000);
updDt();checkAutoSet();
