    <script>
        let selectedCode = ''; let selectedLabel = ''; let currentQty = ''; let timeOffsetMinutes = 0;
        let records = JSON.parse(localStorage.getItem('nippo_records_v6') || '[]');

        window.onload = function() {
            updateTimeDisplay();
            setInterval(updateTimeDisplay, 10000);
            checkAutoSet();
            renderRecords();
        };

        function switchTab(tabId, el) {
            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
            document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
            document.getElementById(tabId).classList.add('active'); el.classList.add('active');
            renderRecords();
        }

        function getTargetDate() {
            let d = new Date(); d.setMinutes(d.getMinutes() + timeOffsetMinutes); return d;
        }

        function updateTimeDisplay() {
            let d = getTargetDate();
            document.getElementById('timeDisplay').innerText = `設定時刻: ${d.toLocaleTimeString('ja-JP', {hour: '2-digit', minute:'2-digit'})} ⏳ (タップで今すぐリセット)`;
        }

        function adjustTime(mins) { timeOffsetMinutes += mins; updateTimeDisplay(); }
        function resetTimeToPresent() { timeOffsetMinutes = 0; updateTimeDisplay(); showToast("現在時刻に戻しました"); }

        function checkAutoSet() {
            let now = new Date(); let hr = now.getHours(); let min = now.getMinutes(); let totalMin = hr * 60 + min;
            if (totalMin >= 510 && totalMin <= 520) {
                selectProcess('W', '朝礼'); currentQty = '0'; document.getElementById('numDisplay').innerText = currentQty; document.getElementById('memoInput').value = '朝礼';
            } else if (hr == 12) {
                selectProcess('Z', 'その他'); currentQty = '0'; document.getElementById('numDisplay').innerText = currentQty; document.getElementById('memoInput').value = '休憩';
            }
        }

        function selectProcess(code, label) {
            selectedCode = code; selectedLabel = label;
            document.querySelectorAll('.btn-proc').forEach(b => b.classList.remove('active'));
            let btn = document.getElementById(`p-${code}`); if(btn) btn.classList.add('active');
            
            // 自動セット時間外に普通にボタンが押されたら文字をクリアする
            let hr = new Date().getHours();
            if(hr != 12 && !(new Date().getHours() == 8 && new Date().getMinutes() <= 40)) {
                if(code !== 'W' && code !== 'Z') { document.getElementById('memoInput').value = ''; }
            }
        }

        function pressKey(key) {
            if (key === 'C') { currentQty = ''; }
            else if (key === 'BS') { currentQty = currentQty.slice(0, -1); }
            else { if(currentQty === '0') currentQty = ''; currentQty += key; }
            document.getElementById('numDisplay').innerText = currentQty;
        }

        function sendQuick() {
            let d = getTargetDate();
            let newRecord = { id: Date.now(), time: d.toLocaleString('ja-JP'), code: 'Z', label: 'その他', qty: 0, memo: 'クイック完了' };
            records.unshift(newRecord); saveAndFeedback("クイック完了を保存");
        }

        function submitData() {
            if(!selectedCode) { alert("工程を選択してください！"); return; }
            let qtyNum = parseInt(currentQty) || 0;
            let d = getTargetDate();

            let newRecord = {
                id: Date.now(),
                time: d.toLocaleString('ja-JP'),
                code: selectedCode,
                label: selectedLabel,
                qty: qtyNum,
                memo: document.getElementById('memoInput').value || ''
            };

            records.unshift(newRecord);
            saveAndFeedback(`スマホに保存しました [${qtyNum}個]`);
            
            currentQty = ''; document.getElementById('numDisplay').innerText = '';
            document.getElementById('memoInput').value = '';
            document.querySelectorAll('.btn-proc').forEach(b => b.classList.remove('active'));
            selectedCode = ''; timeOffsetMinutes = 0; updateTimeDisplay();
        }

        function saveAndFeedback(msg) {
            localStorage.setItem('nippo_records_v6', JSON.stringify(records));
            renderRecords();
            let flash = document.getElementById('flash'); flash.style.display = 'block'; setTimeout(() => flash.style.display = 'none', 150);
            showToast(msg);
        }

        function showToast(msg) {
            let toast = document.getElementById('toast'); toast.innerText = msg; toast.style.display = 'block'; setTimeout(() => toast.style.display = 'none', 2000);
        }

        function editQty(id, oldQty) {
            let newQty = prompt("修正後の数量を入力してください:", oldQty);
            if(newQty !== null) {
                let num = parseInt(newQty) || 0;
                records = records.map(r => r.id === id ? {...r, qty: num} : r);
                localStorage.setItem('nippo_records_v6', JSON.stringify(records));
                renderRecords(); showToast("数量を修正しました");
            }
        }

        function deleteRecord(id) {
            if(confirm("削除しますか？")) {
                records = records.filter(r => r.id !== id);
                localStorage.setItem('nippo_records_v6', JSON.stringify(records));
                renderRecords(); showToast("削除しました");
            }
        }

        function clearAllData() {
            if(confirm("データをすべて消去しますか？\n(Excelに貼った後のリセット用)")) {
                records = []; localStorage.setItem('nippo_records_v6', JSON.stringify(records));
                renderRecords(); showToast("リセットしました");
            }
        }

        function renderRecords() {
            let totalQty = 0;
            let recentDiv = document.getElementById('recentHistory'); recentDiv.innerHTML = '';
            
            records.slice(0, 3).forEach(r => {
                let t = r.time.split(' ')[1] || r.time;
                recentDiv.innerHTML += `
                    <div class="timeline-item">
                        <div class="tl-left">
                            <span class="tl-time">${t}</span>
                            <span class="tl-title">${r.label} (${r.code}) ${r.memo ? `<span class="tl-site">[${r.memo}]</span>` : ''}</span>
                        </div>
                        <div class="tl-right"><span class="tl-qty-click" onclick="editQty(${r.id}, ${r.qty})">${r.qty}個</span></div>
                    </div>`;
            });

            let fullDiv = document.getElementById('fullTimeline'); fullDiv.innerHTML = '';
            if(records.length === 0) {
                fullDiv.innerHTML = '<p style="text-align:center; color:#888; padding:20px;">記録はまだありません</p>';
                document.getElementById('totalQtyDisplay').innerText = "0 個"; return;
            }

            records.forEach(r => {
                if(['A','B','C','D'].includes(r.code)) { totalQty += r.qty; }
                fullDiv.innerHTML += `
                    <div class="timeline-item">
                        <div class="tl-left">
                            <span class="tl-time">${r.time}</span>
                            <span class="tl-title">${r.label} (${r.code})</span><br><span class="tl-site">メモ: ${r.memo || 'なし'}</span>
                        </div>
                        <div class="tl-right">
                            <span class="tl-qty-click" onclick="editQty(${r.id}, ${r.qty})">${r.qty}個</span>
                            <button class="btn-del" onclick="deleteRecord(${r.id})">削除</button>
                        </div>
                    </div>`;
            });
            document.getElementById('totalQtyDisplay').innerText = totalQty + " 個";
        }

        // 🌟コピペデータ作成：数量の右隣（5列目）に自動で「1」を潜り込ませる魔法の関数
        function copyToClipboard() {
            if(records.length === 0) { alert("コピーするデータがありません"); return; }
            let text = "";
            let copyList = [...records].reverse(); // 古い順に並び替え
            
            copyList.forEach(r => {
                let dStr = r.time.split(' ')[0];
                let tStr = r.time.split(' ')[1] ? r.time.split(' ')[1].slice(0,5) : "";
                
                // ➔ 【日時】タブ【自分】タブ【工程記号】タブ【数量】タブ【 1 】タブ【メモ】 の順で1行を作る
                text += `${dStr} ${tStr}\t自分\t${r.code}\t${r.qty}\t1\t${r.memo}\n`;
            });

            navigator.clipboard.writeText(text).then(() => {
                showToast("📋 Excel用に1発コピーしました！シートの左上セルでCtrl+Vしてください");
            }).catch(err => { alert("コピー失敗"); });
        }
    </script>
